function setA(){  // A=X+Y as integers
  var A = parseInt($('#X').text()) + parseInt($('#Y').text());
  $('#A').text( A );
}

setA();
$('#X,#Y').css('cursor','pointer').click(function(){
  // by reaction to a click: updates A and X or Y
  var obj = $(this);
  obj.text(parseInt(obj.text())+1);
  setA();
});
